import { rateLimiter, requireValidToken } from 'qubicshield'; // Deine bestehende Route — eine Zeile Middleware reicht app.get('/api/mein-endpunkt', rateLimiter, // Erkennt Angriffsmuster (>50 req/min) requireValidToken, // Prüft QubicShield-Token beim SC (req, res) => { res.json({ message: 'Hello Qubic — Zugang gewährt' }); } );
